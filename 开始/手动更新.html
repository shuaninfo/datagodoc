# 手动更新

更新文件目录示意

datago ---------------------------------- 应用目录
├── config ------------------------------ 配置文件目录
├── datago ------------------------------ 二进制文件（替换升级）

# 1. 关闭应用

cd /opt/datago
sh stop.sh

# 2. 下载最新版本并解压

cd /home
wget http://static.shuaninfo.com/datago/datago-xxx.tar.gz
tar -zxvf datago-xxx.tar.gz
cd datago-xxx

# 3. 替换二进制文件

cp /home/datago-xxx/datago /opt/datago/datago

# 4. 启动应用

sh start.sh

# 5. 检查日志是否正常，升级完毕

tail -f /opt/datago/run.log
